// Parent class
class Animal {
  constructor(name) {
    this.name = name; // property
  }

  speak() { // method
    console.log(`${this.name} makes a sound`);
  }
}

// Subclass
class Dog extends Animal {
  speak() { // override
    console.log(`${this.name} barks`);
  }
}

// Create objects
const cat = new Animal("Cat");
const dog = new Dog("Rex");

cat.speak(); // Cat makes a sound
dog.speak(); // Rex barks
